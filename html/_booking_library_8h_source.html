187 bool reserve(vector<int> seats) {
188 semaphore.acquire();
189 for (auto seat: seats) {
190 if (this->seats[seat] != SeatStatus::Free) {
191 semaphore.release();
192 return false;
193 }
194 }
195 for (auto seat: seats) {
196 this->seats[seat] = SeatStatus::Reserved;
197 }
198 semaphore.release();
199 return true;
200 }
201
207 bool book(vector<int> seats) {
208 if (reserve(seats)) {
209 for (auto seat: seats) {
210 this->seats[seat] = SeatStatus::Booked;
211 }
212 return true;
213 }
214 return false;
